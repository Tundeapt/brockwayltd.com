// =============================== // PRAXOS ENGINE v1.0 // Dual Pattern + Score System // =============================== // ===== MAIN ENTRY ===== function analyzeMatch(match) { const primary = getPrimaryPatterns(match); const secondary = getSecondaryPatterns(match); const primaryTop = getTopConfidence(primary); const mode = assignMode(primaryTop); const stake = assignStake(primaryTop); const chaos = isChaos(match, primaryTop, secondary); const scores = generateScores(primary, secondary); return formatOutput({ confidence: primaryTop, mode, primary, secondary, scores, stake, chaos }); } // =============================== // PRIMARY PATTERNS (DIRECTION) // =============================== function getPrimaryPatterns(match) { const homeStrength = match.homeForm; const awayWeakness = match.awayForm; const h2h = match.h2h; let p1, p2; if (homeStrength > awayWeakness) { p1 = { label: "1X", confidence: 60 + homeStrength - awayWeakness }; p2 = { label: "1", confidence: 55 + (homeStrength * 0.6) }; } else { p1 = { label: "X2", confidence: 60 + awayWeakness - homeStrength }; p2 = { label: "2", confidence: 55 + (awayWeakness * 0.6) }; } return [normalize(p1), normalize(p2)]; } // =============================== // SECONDARY PATTERNS (BEHAVIOR) // =============================== function getSecondaryPatterns(match) { const goals = match.avgGoals; const bttsRate = match.bttsRate; let s1, s2; if (goals < 2.5) { s1 = { label: "Under 3.5", confidence: 58 }; } else { s1 = { label: "Over 2.5", confidence: 60 }; } if (bttsRate > 0.5) { s2 = { label: "BTTS", confidence: 55 }; } else { s2 = { label: "No BTTS", confidence: 53 }; } return [s1, s2]; } // =============================== // MODE LOGIC // =============================== function assignMode(conf) { if (conf >= 70) return "ATTACK"; if (conf >= 60) return "CONTROL"; if (conf >= 55) return "LOW CONTROL"; return "CHAOS"; } function assignStake(conf) { if (conf >= 70) return "MEDIUM"; if (conf >= 60) return "SMALL"; if (conf >= 55) return "VERY SMALL"; return "NO STAKE"; } // =============================== // CHAOS DETECTION // =============================== function isChaos(match, primaryTop, secondary) { return ( primaryTop < 55 && secondary.every(s => s.confidence < 55) ); } // =============================== // SCORE GENERATOR // =============================== function generateScores(primary, secondary) { const primaryMain = primary[0].label; const secondaryMain = secondary[0].label; let scores = []; // Base from PRIMARY if (primaryMain.includes("1")) { scores = ["2-0", "2-1", "3-0", "3-1"]; } else if (primaryMain.includes("2")) { scores = ["0-1", "1-2", "0-2", "1-3"]; } else { scores = ["1-1", "0-0", "2-2", "1-2"]; } // Adjust with SECONDARY if (secondaryMain.includes("Under")) { scores = scores.filter(s => totalGoals(s) <= 3); } if (secondaryMain.includes("BTTS")) { scores = scores.filter(s => bothTeamsScore(s)); } // Ensure at least 4 while (scores.length < 4) { scores.push("1-1"); } scores = scores.slice(0, 4); return distributeScoreConfidence(scores, secondary[0].confidence); } // =============================== // HELPERS // =============================== function totalGoals(score) { const [a, b] = score.split("-").map(Number); return a + b; } function bothTeamsScore(score) { const [a, b] = score.split("-").map(Number); return a > 0 && b > 0; } function distributeScoreConfidence(scores, base) { const weights = [0.30, 0.27, 0.23, 0.20]; return scores.map((s, i) => ({ score: s, confidence: Math.round(base * weights[i]) })); } function getTopConfidence(arr) { return Math.max(...arr.map(a => a.confidence)); } function normalize(obj) { obj.confidence = Math.min(75, Math.max(50, Math.round(obj.confidence))); return obj; } // =============================== // FINAL OUTPUT FORMAT // =============================== function formatOutput(data) { if (data.chaos) { return { Confidence: data.confidence + "%", Mode: "CHAOS", Primary: data.primary, Secondary: data.secondary, Scores: data.scores, Stake: "NO STAKE" }; } return { Confidence: data.confidence + "%", Mode: data.mode, Primary: data.primary, Secondary: data.secondary, ScorePatterns: data.scores, Stake: data.stake }; }
