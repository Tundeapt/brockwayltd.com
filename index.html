calculate_raw_performance(graded_matches: list) -> dict: performance = {} for match in graded_matches: for s_tip in match.get("success_list", []): tip = s_tip.upper() if tip not in performance: team (away)": win = (ag > hg) elif tip_lower == "draw": win = (hg == ag) elif tip_lower == "over 1.5": win = (hg + ag > 1.5) elif tip_lower == "over 2.5": win = (hg + ag > 2.5) elif tip_lower == "over 3.5": win = (hg + ag > 3.5) elif tip_lower == "under 1.5": win = (hg + ag < 1.5) elif tip_lower == "under 2.5": win = (hg + ag < 2.5) elif tip_lower == "under 3.5": win = (hg + ag < 3.5) elif tip_lower == "btts (yes)": win = (hg > 0 and ag > 0) elif tip_lower == "btts (no)": win = (hg == 0 or ag == 0) elif tip_lower == "win or draw (1x)": win = (hg >= ag) elif tip_lower == "win or draw (x2)": win = (ag >= hg) elif tip_lower == "home or away to win (12)": win = (hg != ag) elif tip_lower == "home over 0.5": win = (hg > 0) elif tip_lower == "away over 0.5": win = (ag > 0) elif tip_lower == "home over 1.5": win = (hg > 1) elif tip_lower == "away over 1.5": win = (ag > 1) elif tip_lower == "draw or over 2.5": win = (hg == ag) or ((hg + ag) > 2.5) elif tip_lower == "home handicap (+1.5)": win = (hg + 1.5 > ag) elif tip_lower == "home handicap (-1.5)": win = (hg - 1.5 > ag) elif tip_lower == "away handicap (+1.5)": win = (ag + 1.5 > hg) elif tip_lower == "away handicap run_autograder(pending_list): if not pending_list: return 0 graded_count = 0 headers = {"x-apisports-key": FOOTBALL_KEY} # Process in batches of 20 for i in range(0, len(pending_list), 20): batch = timeout=15) if resp.status_code != 200: continue for match_info in resp.json().get("response", []): m_id = str(match_info["fixture"]["id"]) status = match_info["fixture"]["status"]["short"] if status in ["FT", "AET", "PEN"]: hg = match_info.get("goals", {}).get("home") ag = match_info.get("goals", {}).get("away") if hg is None or ag is None: continue # Find matching tips target_match = next((m for supabase.table("praxos_absolute").update({ "actual_score": f"{hg} - {ag}", "success_list": success, "fail_list": fail, "fixture_grade": grade, "grade_status": "graded" }).eq("match_id", build_accumulators(raw_pool_data, target_legs, max_hours_span, target_ticket_count, db_killed_tips): current_utc_now = datetime.now(timezone.utc) broker_safety_buffer = current_utc_now + = [] completed_tickets = [] global_killed_tips = set(db_killed_tips) for tip in filtered_pool: if len(completed_tickets) >= target_ticket_count: break unique_tip_id = f"{tip['match_id']}_{tip['tip']}" if unique_tip_id in global_killed_tips: continue placed = False tip_time = tip['kickoff_ts'] for ticket in active_tickets: # Collision Avoidance: Max 1 tip per match per ticket if any(leg['match_id'] == tip['match_id'] for leg in ticket['legs']): continue earliest = min(ticket['earliest_ts'], tip_time) latest = max(ticket['latest_ts'], tip_time) time_span_hours = (latest - earliest).total_seconds() / 3600 if time_span_hours > max_hours_span: continue ticket['legs'].append(tip) ticket['earliest_ts'] = earliest ticket['latest_ts'] = latest global_killed_tips.add(unique_tip_id) placed = True if len(ticket['legs']) == target_legs: completed_tickets.append(ticket) active_tickets.remove(ticket) break if not placed: active_tickets.append({ 'ticket_id': len(completed_tickets) + len(active_tickets) + 1, 'legs': [tip], 'earliest_ts': tip_time, 'latest_ts': tip_time }) global_killed_tips.add(unique_tip_id) # Pad remaining tickets if len(completed_tickets) < target_ticket_count and active_tickets: active_tickets.sort(key=lambda x: len(x['legs']), reverse=True) for t in active_tickets: if len(completed_tickets) >= target_ticket_count: break if len(t['legs']) > 0: completed_tickets.append(t) completed_tickets.sort(key=lambda x: x['earliest_ts']) return completed_tickets #
